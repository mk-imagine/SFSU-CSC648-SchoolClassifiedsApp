\section{Main Data Items and Entities}

\begin{enumerate}

\item \textbf{user}
\begin{description}[font=\normalfont\itshape]
	\item[first\_name:] First name of user
	\item[last\_name:] Last name of user
	\item[email:] Email of user
\end{description}
	
\item \textbf{item}
\begin{description}[font=\normalfont\itshape]
	\item[item\_name:] Name of item
	\item[item\_description:] Item description
	\item[item\_price:] Item price
	\item[item\_category:] Item category
\end{description}
	
\item \textbf{category}
\begin{description}[font=\normalfont\itshape]
	\item[cat\_name:] Category name
\end{description}

\item \textbf{account}
\begin{description}[font=\normalfont\itshape]
	\item[acct\_type:] Account type (e.g. admin, registered\_user, etc)
	\item[acct\_privileges:] Account privileges
\end{description}
	
\item \textbf{location}
\begin{description}[font=\normalfont\itshape]
	\item[loc\_name:] Location name
	\item[loc\_description:] Location description
	\item[loc\_coords:] Coordinates for location
\end{description}
	
\item \textbf{message}
\begin{description}[font=\normalfont\itshape]
	\item[message\_sender:] Message sender
	\item[message\_recipient:] Message recipient
	\item[message\_contents:] Contents of message
\end{description}
	
\item \textbf{review}
\begin{description}[font=\normalfont\itshape]
	\item[reviewed:] User being reviewed
	\item[reviewer:] User reviewing
	\item[review\_details:] Contents of review
\end{description}

\end{enumerate}
